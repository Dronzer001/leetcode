class Solution {
public:
    vector<int> decimalRepresentation(int n) {
        vector<int>ans;
        int i=0;
        while(n!=0){
            int mid=n%10  *pow(10,i);
            if(mid!=0) ans.push_back(mid);
            n/=10;
            i+=1;
        }
        reverse(ans.begin(),ans.end());
        return ans;
    }
};
